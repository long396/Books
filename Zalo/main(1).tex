\parbox{17cm}{
    \fbox{
        \parbox{17cm}{
            Tính xấp xỉ nghiệm hệ      
            $$y''(x_1) + p(x_i)y'(x_1) + q(x_i)y(x_1) = r(x_1)$$
            $$y(x_0) = \alpha; y'(x_N) = \beta$$
        }
    } 
    \fbox{
        \parbox{17cm}{
            Ta chỉ làm cách này được nếu $N \geq 5$
        }
    }
    \fbox{
        \parbox{17cm}{
            Với $i=1$:\\
            \[y''(x_1) + p(x_i)y'(x_1) + q(x_i)y(x_1) = r(x_1)\]
            $$
            \frac{y(x_2) - 2y(x_1) + y(x_0)}{h^2}
            + p(x)\frac{y(x_2) - y(x_0)}{2h}
            + q(x)y(x_i) = r(x_i) + \mathcal{O}(h^2)
            $$
            $$
              y(x_0)\left[\frac{1}{h^2} - \frac{p(x_1)}{2h}\right]
            + y(x_1)\left[\frac{-2}{h^2} + q(x_1)\right]
            + y(x_2)\left[\frac{1}{h^2} + \frac{p(x_1)}{2h}\right]
            = r(x_1) + \mathcal{O}(h^2)
            $$
            $$
              y(x_1)\left[\frac{-2}{h^2} + q(x_1)\right]
            + y(x_2)\left[\frac{1}{h^2} + \frac{p(x_1)}{2h}\right]
            = r(x_1) - y(x_0)\left[\frac{1}{h^2} - \frac{p(x_1)}{2h}\right] + \mathcal{O}(h^2)
            $$
            $$
              y(x_1)\left[\frac{-2}{h^2} + q(x_1)\right]
            + y(x_2)\left[\frac{1}{h^2} + \frac{p(x_1)}{2h}\right]
            = r(x_1) - \alpha\left[\frac{1}{h^2} - \frac{p(x_1)}{2h}\right] + \mathcal{O}(h^2)
            $$
        }
    }\\
    \fbox{
        \parbox{17cm}{
            Với $2 \leq i \leq N-2$:\\
            $$y''(x_i) + p(x_i)y'(x_i) + q(x_i)y(x_i) = r(x_i)$$\\
            $$
            \frac{y(x_{i+1}) - 2y(x_i) + y(x_{i-1})}{h^2}
            + p(x)\frac{y(x_{i+1}) - y(x_{i-1})}{2h}
            + q(x)y(x_i) = r(x_i) + \mathcal{O}(h^2)
            $$
            $$
              y(x_{i-1})\left[\frac{1}{h^2} - \frac{p(x_i)}{2h}\right]
            + y(x_{i})\left[\frac{-2}{h^2} + q(x_i)\right]
            + y(x_{i+1})\left[\frac{1}{h^2} + \frac{p(x_i)}{2h}\right]
            = r(x_i) + \mathcal{O}(h^2)
            $$
            
        }
    }
    \fbox{
        \parbox{17cm}{
        \tiny
            Với $i = N-1$:\\
            $$
            y''(x_{N-1}) + p(x_{N-1})y'(x_{N-1}) + q(x_{N-1})y(x_{N-1}) = r(x_{N-1})
            $$
            $$
            \frac{y'(x_{N}) - y'(x_{N-2})}{2h}
            + p(x_{N-1})\frac{3y(x_{N-1})-4y(x_{N-2})+y(x_{N-3})}{2h}
            + q(x_{N-1})y(x_i) = r(x_i) + \mathcal{O}(h^2)
            $$
            \[
            \frac{y'(x_{N}) - \frac{11y(x_{N-2})-18y(x_{N-3})+9y(x_{N-4})-2y(x_{N-5})}{h}}{2h}
            + p(x_{N-1})\frac{3y(x_{N-1})-4y(x_{N-2})+y(x_{N-3})}{2h}\\
            + q(x_{N-1})y(x_{N-1}) = r(x_{N-1}) + \mathcal{O}(h^2)
            \]
            \[
            \frac{\beta}{2h}- \frac{11y(x_{N-2})-18y(x_{N-3})+9y(x_{N-4})-2y(x_{N-5})}{2h^2}
            + p(x_{N-1})\frac{3y(x_{N-1})-4y(x_{N-2})+y(x_{N-3})}{2h}\\
            + q(x_{N-1})y(x_{N-1}) = r(x_{N-1}) + \mathcal{O}(h^2)
            \]
            \[
              y(x_{N-5})\left[\frac{1}{h^2}\right]
            + y(x_{N-4})\left[\frac{-9}{2h^2}\right]
            + y(x_{N-3})\left[\frac{9}{h^2}+\frac{p(x_{N-1})}{2h}\right]
            + y(x_{N-2})\left[\frac{11}{2h^2}-\frac{4p(x_{N-1})}{2h}\right]
            + y(x_{N-1})\left[q(x_{N-1})+\frac{3p(x_{N-1})}{2h}\right]
            = r(x_{N-1}) - \frac{\beta}{2h} + \mathcal{O}(h^2)
            \]
        }
    }
}
